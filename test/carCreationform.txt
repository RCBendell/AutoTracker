def createCar(request):
   if request.method == 'POST':
      form = CarCreationForm(request.POST)
      if form.is_valid():
         form.save()
         return redirect('profile')
   else:
      form = CarCreationForm()
   return render(request, 'car_creation.html', {'form': form})
